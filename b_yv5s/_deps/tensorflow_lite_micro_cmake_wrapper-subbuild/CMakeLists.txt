cmake_minimum_required(VERSION 3.27.9)

# Reject any attempt to use a toolchain file. We must not use one because
# we could be downloading it here. If the CMAKE_TOOLCHAIN_FILE environment
# variable is set, the cache variable will have been initialized from it.
unset(CMAKE_TOOLCHAIN_FILE CACHE)
unset(ENV{CMAKE_TOOLCHAIN_FILE})

# We name the project and the target for the ExternalProject_Add() call
# to something that will highlight to the user what we are working on if
# something goes wrong and an error message is produced.

project(tensorflow_lite_micro_cmake_wrapper-populate NONE)



include(ExternalProject)
ExternalProject_Add(tensorflow_lite_micro_cmake_wrapper-populate
                     "UPDATE_DISCONNECTED" "False" "EXTERNALPROJECT_INTERNAL_ARGUMENT_SEPARATOR" "URL" "https://gitlab.arm.com/artificial-intelligence/ethos-u/ethos-u-core-software/-/raw/25.02/tflite_micro.cmake?ref_type=tags&inline=false" "URL_HASH" "MD5=7ee273b9d993d8552f7d1fc129f276fd" "DOWNLOAD_DIR" "/home/linzejia/app/mlek_lee/b_yv5s" "DOWNLOAD_NAME" "tflite_micro.cmake" DOWNLOAD_NO_EXTRACT YES
                    SOURCE_DIR          "/home/linzejia/app/mlek_lee/b_yv5s/_deps/tensorflow_lite_micro_cmake_wrapper-src"
                    BINARY_DIR          "/home/linzejia/app/mlek_lee/b_yv5s/_deps/tensorflow_lite_micro_cmake_wrapper-build"
                    CONFIGURE_COMMAND   ""
                    BUILD_COMMAND       ""
                    INSTALL_COMMAND     ""
                    TEST_COMMAND        ""
                    USES_TERMINAL_DOWNLOAD  YES
                    USES_TERMINAL_UPDATE    YES
                    USES_TERMINAL_PATCH     YES
)


ExternalProject_Get_Property(tensorflow_lite_micro_cmake_wrapper-populate DOWNLOADED_FILE)
get_filename_component(dlFileName "${DOWNLOADED_FILE}" NAME)

ExternalProject_Add_Step(tensorflow_lite_micro_cmake_wrapper-populate copyfile
  COMMAND    "/home/linzejia/app/cmake/bin/cmake" -E copy_if_different
             "<DOWNLOADED_FILE>" "/home/linzejia/app/mlek_lee/b_yv5s/_deps/tensorflow_lite_micro_cmake_wrapper-src"
  DEPENDEES  patch
  DEPENDERS  configure
  BYPRODUCTS "/home/linzejia/app/mlek_lee/b_yv5s/_deps/tensorflow_lite_micro_cmake_wrapper-src/${dlFileName}"
  COMMENT    "Copying file to SOURCE_DIR"
)
